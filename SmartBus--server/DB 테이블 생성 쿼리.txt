본 쿼리는 MYSQL WorkBench에서 사용함.


1. USER 테이블
CREATE TABLE `user` (
  `IMEI_ID` varchar(20) NOT NULL,
  PRIMARY KEY (`IMEI_ID`)
) 

2. 즐겨찾기 테이블
CREATE TABLE `bookmark` (
  `USER_IMEI` varchar(20) NOT NULL,
  `DEPTNODE` varchar(30) NOT NULL,
  `DESTNODE` varchar(30) NOT NULL,
  `TRANSNODE1` varchar(30) DEFAULT NULL,
  `TRANSNODE2` varchar(30) DEFAULT NULL,
  `ROUTENO1` varchar(10) NOT NULL,
  `TROUTENO1` varchar(10) DEFAULT NULL,
  `TROUTENO2` varchar(10) DEFAULT NULL,
  KEY `FK_BOOKMARK_IMEI_idx` (`USER_IMEI`),
  CONSTRAINT `FK_BOOKMARK_IMEI` FOREIGN KEY (`USER_IMEI`) REFERENCES `user` (`IMEI_ID`) ON DELETE CASCADE ON UPDATE CASCADE
) 


3. 최근기록 테이블
CREATE TABLE `recent` (
  `USER_IMEI` varchar(20) NOT NULL,
  `DEPTNODE` varchar(30) NOT NULL,
  `DESTNODE` varchar(30) NOT NULL,
  `TRANSNODE1` varchar(30) DEFAULT NULL,
  `TRANSNODE2` varchar(30) DEFAULT NULL,
  `ROUTENO1` varchar(10) NOT NULL,
  `TROUTENO1` varchar(10) DEFAULT NULL,
  `TROUTENO2` varchar(10) DEFAULT NULL,
  KEY `FK_RECENT_IMEI_idx` (`USER_IMEI`),
  CONSTRAINT `FK_RECENT_IMEI` FOREIGN KEY (`USER_IMEI`) REFERENCES `user` (`IMEI_ID`) ON DELETE CASCADE ON UPDATE CASCADE
)
